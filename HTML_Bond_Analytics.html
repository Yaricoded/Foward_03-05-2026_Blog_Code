#Core Bond Math 

def build_cash_flows(face, coupon_rate, years, freq):
    n = round(years * freq)
    c = (coupon_rate/100) * face / freq
    periods = np.arange(1, n+1)
    coupons = np.full(n, c)
    principal = np.zeros(n)
    principal[-1] = face
    times = periods / freq
    return pd.DataFrame({'period': periods, 'time': times, 'coupon': coupons, 'principal': principal})

def price_from_yield(flows, ytm_annual, freq):
    r = ytm_annual/100/freq
    cf = flows['coupon'] + flows['principal']
    df = (1+r) ** (-flows['period'])
    return float((cf*df).sum())

def ytm_from_price(flows, target_price, freq, guess=5.0):
    lo, hi = 1e-4, 100.0
    def f(y): return price_from_yield(flows, y, freq) - target_price
    if f(lo)*f(hi) > 0:
        return None
    y = guess
    for _ in range(60):
        fy = f(y)
        if abs(fy) < 1e-7:
            return y
        h = 1e-4
        dfy = (f(y+h) - f(y-h)) / (2*h)
        y_next = y - fy/dfy if dfy != 0 else (lo+hi)/2
        if not np.isfinite(y_next) or y_next <= 0 or y_next > 100:
            mid = (lo+hi)/2
            if f(lo)*f(mid) <= 0:
                hi = mid
            else:
                lo = mid
            y_next = (lo+hi)/2
        else:
            if fy > 0:
                lo = y
            else:
                hi = y
        y = y_next
    return y

def macaulay_duration(flows, ytm_annual, freq, price):
    r = ytm_annual/100/freq
    cf = flows['coupon'] + flows['principal']
    df = (1+r) ** (-flows['period'])
    weighted_t = (flows['time']*cf*df).sum()
    return weighted_t/price

def modified_duration(mac_d, ytm_annual, freq):
    r = ytm_annual/100/freq
    return mac_d/(1+r)

def convexity(flows, ytm_annual, freq, price):
    r = ytm_annual/100/freq
    cf = flows['coupon'] + flows['principal']
    df = (1+r) ** (-flows['period'])
    s = (cf*df*flows['period']*(flows['period']+1)).sum()
    convex_periods = s/(price*(1+r)**2)
    return convex_periods/(freq**2)

flows = build_cash_flows(FACE_VALUE, COUPON_RATE, YEARS, FREQ)

#Duration & Convexity approximation check
shocks = np.arange(-3, 3.01, 0.25)  #percentage point yield shocks

actual = np.array([price_from_yield(flows, MARKET_YIELD+s, FREQ) for s in shocks])
dur_only = price - mod_d*price*(shocks/100)
dur_conv = price - mod_d*price*(shocks/100) + 0.5*conv*price*(shocks/100)**2

fig, axes = plt.subplots(1, 2, figsize=(14, 5.5))

axes[0].plot(shocks, actual, color=BLUE_DEEP, linewidth=2.5, label='Actual repriced')

8. Summary¶

  • Duration measures the first order (linear) sensitivity of price to yield changes: it's accurate for small yield moves but systematically underestimates price gains on yield drops and overestimates price losses on yield rises, because the true price yield relationship is convex, not linear.
  • Convexity is the second order correction: adding it brings the estimated price curve much closer to the true repriced bond, especially for larger yield shocks (±2-3 percentage points).
  • The YTM solver (Newton Raphson + bisection fallback) recovers the discount rate implied by any observed market price, which is the standard way bond yields are quoted from prices in practice.

Try changing COUPON_RATE, YEARS, FREQ, or MARKET_YIELD / MARKET_PRICE in the cells above and rerunning to see how the price, duration, and convexity shift for a different bond.
